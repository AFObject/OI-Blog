F 琪露诺的选择题 - P8471

Medium构造容斥

算法

设正确答案里有 \(x\) 个 A。

我们只能放 \(a\) 个 A，现在考虑放几个在正确的位置上。假设放 \(u\) 个。那么在错误位置上的就是 \(a-u\) 个。

放在正确位置上的有 \(u\) 个，那么剩下的 \(x-u\) 个正确答案里的 A 就得对应 B。对最终答案（\(e\)）的贡献，要么是 A 放在了 B 上（\(a-u\)），要么是 B 放在了 A 上（\(x-u\)）。因此 \(e=x-u+a-u=x+a-2u\)。好的，\(u\) 算出来了，是 \(\dfrac{x+a-e}2\)。

不能解的情况，就是算出来的 \(u>x\)，或者 \(u>a\)。

实现

多测给我清空！!1多测给我清空！!1多测给我清空！!1多测给我清空！!1多测给我清空！!1多测给我清空！!1

int T;
int n, a, e;
char s[300001], ans[300001];

int main() {
    T = read();
    while (T--) {
        n = read() << 1, a = read(), e = read();
        scanf("%s", s + 1);
        int x = 0;
        for (int i = 1; i <= n; ++i)
            x += (s[i] == 'A');
        int u = (x + a - e) >> 1, v = a - u;
        if (v > n - x || u < 0 || u > x || u > a || (x + a - e) & 1) {
            puts("-1");
            continue;
        }
        memset(ans, 0, sizeof(ans));
        for (int i = 1; i <= n; ++i)
            ans[i] = 'B';
        for (int i = 1, t = 1; t <= u; ++i)
            if (s[i] == 'A')
                ans[i] = 'A', ++t;
        for (int i = 1, t = 1; t <= a - u; ++i)
            if (s[i] == 'B')
                ans[i] = 'A', ++t;
        puts(ans + 1);
    }
    return 0;
}

H 破碎的历史 - P8473

MediumDS线段树

算法

每次操作取最有用的一条线段（最左端第一个黑到最右端第一个黑）。并记录。每次线段树将区间内的数都加一，每次统计是否全部 \(\geq 1\)。

实现

#define lt(x) ((x) << 1)
#define rt(x) (((x) << 1) + 1)

int n, q;
int a[500001], c[500001];

namespace SegTree {
    int tag[2000001], min[2000001];
    void split(int x) { // 需要 lazytag 捏
        tag[lt(x)] += tag[x], tag[rt(x)] += tag[x];
        min[lt(x)] += tag[x], min[rt(x)] += tag[x];
        min[x] += tag[x], tag[x] = 0;
    }
    void merge(int x) {
        min[x] = std::min(min[lt(x)], min[rt(x)]);
    }
    void build(int x, int l, int r) {
        if (l == r) {
            min[x] = c[l];
            return;
        }
        int mid = (l + r) >> 1;
        build(lt(x), l, mid);
        build(rt(x), mid + 1, r);
        merge(x);
    }
    void add(int x, int l, int r, int ql, int qr, int d) {
        if (ql <= l && r <= qr) {
            tag[x] += d;
            min[x] += d;
            return;
        }
        split(x);
        int mid = (l + r) >> 1;
        if (ql <= mid) add(lt(x), l, mid, ql, qr, d);
        if (qr > mid) add(rt(x), mid + 1, r, ql, qr, d);
        merge(x);
    }
}

int l[500001], r[500001]; // 这块儿的用处是定位询问区间内两端的特殊点
int mapL(int x) {
    return std::lower_bound(a + 1, a + n + 1, x) - a;
}
int mapR(int x) {
    return (std::upper_bound(a + 1, a + n + 1, x) - a) - 1;
}

int bl[500001], br[500001]; // 预处理一下每个特殊点左右两端的最近的黑点
void go() {
    int last = n + 1;
    for (int i = n; i; --i) {
        if (c[i]) last = i;
        bl[i] = last;
    }
    last = 0;
    for (int i = 1; i <= n; ++i) {
        if (c[i]) last = i;
        br[i] = last;
    }
}

int main() {
    n = read(), q = read();
    for (int i = 1; i <= n; ++i)
        a[i] = read();
    for (int i = 1; i <= n; ++i)
        c[i] = read();

    go();
    SegTree::build(1, 1, n);

    for (int i = 1; i <= q; ++i) {
        puts(SegTree::min[1] ? "Yes" : "No");
        int op = read();
        if (op == 1) {
            int x = read(), y = read();
            l[i] = mapL(x), r[i] = mapR(y);
            SegTree::add(1, 1, n, bl[l[i]], br[r[i]], 1);
        } else {
            int x = read();
            SegTree::add(1, 1, n, bl[l[x]], br[r[x]], -1);
        }
    }
    puts(SegTree::min[1] ? "Yes" : "No");
    return 0;
}
