当一个右括号把一个左括号抵掉的时候，答案就是右括号对应的下标减去抵掉后栈顶的下标。

实现

int main() {
    scanf("%s", s + 1);
    const int n = strlen(s + 1);
    int max = 0;
    for (int i = 1; i <= n; ++i) {
        if (s[i] == '(' || s[i] == '[' || s[i] == '{') {
            q.push(i);
        } else {
            if (q.empty()) continue;
            char c = s[q.top()];
            if (s[i] == ')' && c == '('
             || s[i] == ']' && c == '['
             || s[i] == '}' && c == '{') {
                q.pop();
                if (q.size()) max = std::max(max, i - q.top());
                else max = std::max(max, i);
            } else {
                q.push(i);
            }
        }
    }
    printf("%d\n", max);
}

int power(int x, int y) {
    int ans = 1;
    while (y--) ans *= x;
    return ans;
}

int cal(int l, int r) {
    char op = '%';
    int q = 0, add = -1, mul = -1, pow = -1;
    for (int i = l; i <= r; ++i) {
        if (s[i] == '(') ++q;
        if (s[i] == ')') --q;
        if (!q && (s[i] == '+' || s[i] == '-')) add = i;
        if (!q && (s[i] == '*' || s[i] == '/')) mul = i;
        if (!q && s[i] == '^') pow = i;
    }
    if (add != -1) {
        if (s[add] == '+') return cal(l, add - 1) + cal(add + 1, r);
        else return cal(l, add - 1) - cal(add + 1, r);
    } else if (mul != -1) {
        if (s[mul] == '*') return cal(l, mul - 1) * cal(mul + 1, r);
        else return cal(l, mul - 1) / cal(mul + 1, r);
    } else if (pow != -1) {
        return power(cal(l, pow - 1), cal(pow + 1, r));
    } else if (s[l] == '(' && s[r] == ')') 
        return cal(l + 1, r - 1);
    else {
        int num = 0;
        for (int i = l; i <= r; ++i)
            num *= 10, num += s[i] - '0';
        return num;
    }
}

City Game

0x1803Medium-栈单调栈

算法

确立一条边为底边，然后按上面这么做。

实现

int main() {
    std::cin >> n >> m;
    for (int i = 1; i <= n; ++i)
        for (int j = 1; j <= m; ++j) {
            char c;
            std::cin >> c;
            if (c == 'R') up[i][j] = 0;
            else up[i][j] = up[i - 1][j] + 1;
        }
    int ans = 0;
    for (int d = 1; d <= n; ++d) {
        q = std::stack<std::pair<int, int>>();
        for (int i = 1; i <= m + 1; ++i) {
            int x = up[d][i], w = 0;
            while (q.size() && q.top().first >= x) {
                w += q.top().second;
                ans = std::max(ans, 3 * q.top().first * w);
                q.pop();
            }
            q.push({up[d][i], w + 1});
        }
    }
    printf("%d\n", ans);
}

实现

int main() {
    n = read(), k = read();
    for (int i = 1; i <= n; ++i) {
        a[i] = read();
        while (q1.size() && a[q1.back()] > a[i]) q1.pop_back();
        q1.push_back(i); 
        while (q1.size() && q1.front() <= i - k) q1.pop_front();

        while (q2.size() && a[q2.back()] < a[i]) q2.pop_back();
        q2.push_back(i);
        while (q2.size() && q2.front() <= i - k) q2.pop_front();

        min[i] = a[q1.front()];
        max[i] = a[q2.front()];
    }
    for (int i = k; i <= n; ++i)
        printf("%d ", min[i]);
    puts("");
    for (int i = k; i <= n; ++i)
        printf("%d ", max[i]);
    puts("");
}
